**Auto-sync on mount (D-10):**
Use an `Effect` that fires once when `dashboard_resource` first resolves:
```rust
let auto_synced = RwSignal::new(false);
Effect::new(move || {
    if let Some(Ok(data)) = dashboard_resource.get() {
        if data.should_auto_sync && data.has_puuid && !auto_synced.get_untracked() {
            auto_synced.set(true);
            syncing.set(true);
            leptos::task::spawn_local(async move {
                match sync_solo_matches().await {
                    Ok(n) => {
                        toast.show.run((ToastKind::Success, format!("Synced {n} new matches")));
                        dashboard_resource.refetch();
                    }
                    Err(e) => toast.show.run((ToastKind::Error, format!("{e}"))),
                }
                syncing.set(false);
            });
        }
    }
});
```

let filtered_matches = move || {
