1. **Create `get_opponent_intel_full` server fn** in draft.rs (or rename/replace `get_opponent_intel`):
   ```rust
   #[server]
   pub async fn get_opponent_intel_full(
       opponent_id: String,
   ) -> Result<Vec<OpponentPlayerIntel>, ServerFnError> {
       // Inside #[server] body:
       use crate::server::db;
       use crate::server::riot;
       use crate::server::data_dragon;
       // ... auth guard, db context ...

       // 1. Fetch opponent players (existing db::get_opponent_players or similar)
       let players = db::get_opponent_players(&db, &opponent_id).await
           .map_err(|e| ServerFnError::new(e.to_string()))?;

       // 2. Build champion_id -> name map from Data Dragon
       let champion_map = build_champion_id_map(); // HashMap<i64, String>

       // 3. For each player, compute intel:
       let mut results = Vec::new();
       for player in players {
           // Frequency counts from recent_champions
           let mut freq_map: HashMap<String, u32> = HashMap::new();
           for champ in &player.recent_champions {
               *freq_map.entry(champ.clone()).or_insert(0) += 1;
           }
           let total_games = player.recent_champions.len() as u32;
           let mut champion_frequencies: Vec<(String, u32)> = freq_map.into_iter().collect();
           champion_frequencies.sort_by(|a, b| b.1.cmp(&a.1));

           // OTP detection: most-played champion > 60% of total games
           let otp_champion = champion_frequencies.first()
               .and_then(|(name, count)| {
                   if total_games > 0 && (*count as f64 / total_games as f64) > 0.60 {
                       Some(name.clone())
                   } else {
                       None
                   }
               });

           // Mastery data (graceful degradation)
           let mastery_data = if riot::has_api_key() {
               if let Some(puuid) = &player.riot_puuid {
                   match riot::fetch_champion_masteries(puuid).await {
                       Ok(masteries) => {
                           masteries.into_iter()
                               .take(10) // top 10 mastery champions
                               .filter_map(|(champ_id, level, points)| {
                                   champion_map.get(&champ_id)
                                       .map(|name| (name.clone(), level, points))
                               })
                               .collect()
                       }
                       Err(_) => Vec::new(), // graceful: no mastery on API error
                   }
               } else {
                   Vec::new() // no puuid linked
               }
           } else {
               Vec::new() // no API key configured
           };

           results.push(OpponentPlayerIntel {
               player,
               champion_frequencies,
               mastery_data,
               otp_champion,
           });
       }
       Ok(results)
   }
   ```
