pub async fn set_user_mode(db: &Surreal<Db>, user_id: &str, mode: &str) -> DbResult<()> {
    let user_key = user_id.strip_prefix("user:").unwrap_or(user_id).to_string();
    db.query("UPDATE type::record('user', $user_key) SET mode = $mode")
        .bind(("user_key", user_key))

}

              # Check each hit has a focus-visible:ring within 3 lines of context

| T-17-01 | Tampering | `set_user_theme(theme: String)` server fn | mitigate | Server fn pre-validates `theme IN ['demacia','pandemonium']` (fast fail); SurrealDB ASSERT enforces same constraint defense-in-depth (schema.surql) |
| T-17-02 | Information Disclosure | `<html data-theme>` attribute | accept | Theme name is non-sensitive (no PII); revealing `demacia` vs `pandemonium` to an unauth visitor leaks nothing |
