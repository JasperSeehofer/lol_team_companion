**Helper function** for event tooltip text:
```rust
fn event_tooltip(event: &TimelineEvent, participants: &[MatchParticipant]) -> String {
    let time_min = event.timestamp_ms / 60_000;
    let time_sec = (event.timestamp_ms % 60_000) / 1000;
    let time_str = format!("{time_min}:{time_sec:02}");
    match event.event_type.as_str() {
        "ELITE_MONSTER_KILL" => {
            let monster = event.monster_type.as_deref().unwrap_or("Monster");
            let sub = event.monster_sub_type.as_deref().map(|s| format!(" ({s})")).unwrap_or_default();
            format!("{time_str} - {monster}{sub} killed")
        }
        "BUILDING_KILL" => {
            let building = event.building_type.as_deref().unwrap_or("Building");
            format!("{time_str} - {building} destroyed")
        }
        "CHAMPION_KILL" => {
            let killer = event.killer_participant_id
                .and_then(|id| participants.iter().find(|p| p.participant_id == id))
                .map(|p| p.summoner_name.as_str()).unwrap_or("Unknown");
            let victim = event.victim_participant_id
                .and_then(|id| participants.iter().find(|p| p.participant_id == id))
                .map(|p| p.summoner_name.as_str()).unwrap_or("Unknown");
            let mut text = format!("{time_str} - {killer} killed {victim}");
            if event.is_first_blood { text.push_str(" (First Blood)"); }
            if let Some(mk) = event.multi_kill_length {
                if mk >= 2 { text.push_str(&format!(" ({mk}x kill)")); }
            }
            text
        }
        "WARD_PLACED" => format!("{time_str} - Ward placed"),
        "TEAMFIGHT" => format!("{time_str} - Teamfight ({} players)", event.involved_participants.len()),
        _ => format!("{time_str} - {}", event.event_type),
    }
}
```

// Bar track
<div class="relative w-full h-10 bg-elevated border border-divider rounded-lg">
    // Render visible events
    {move || {

            let left_pct = timeline_pct(event.timestamp_ms, game_dur);
            let tooltip = event_tooltip(event, &participants_for_timeline);

            // Marker styling by event type (per UI-SPEC TimelineBar):
            let (size_class, shape_class) = match event.category {
                EventCategory::Objective => {
                    match event.monster_type.as_deref() {
                        Some(m) if m.contains("BARON") || m.contains("HORDE") =>
                            ("w-4 h-4", "rounded-full border-2"), // 16px, baron/herald

                    }

                EventCategory::Tower => ("w-2 h-2", "rounded-sm"), // 8px square
                EventCategory::Kill => ("w-2 h-2", "rounded-full"), // 8px circle
                EventCategory::Ward => ("w-2 h-2", "rounded-full"), // 8px dot
                EventCategory::Recall => ("w-2 h-2", "rounded-full"), // 8px circle, per D-07
                EventCategory::Teamfight => ("w-4 h-4", "rounded-full"), // 16px
            };

            // Team color
            let team_color = match event.team_id {
                Some(100) => "bg-blue-400 border-blue-500",
                Some(200) => "bg-red-400 border-red-500",
                _ => "bg-muted border-muted",
            };

            // User's own events get accent ring (per D-09)
            let is_user_event = event.killer_participant_id == Some(user_participant_id)
                || event.involved_participants.contains(&user_participant_id);
            let user_ring = if is_user_event {
                " ring-2 ring-accent ring-offset-1 ring-offset-base"
            } else { "" };

            // Selected state
            let is_selected = selected_event.get() == Some(idx);
            let selected_class = if is_selected { " scale-150 z-20" } else { " z-10" };

            view! {
                <button
                    class=format!("absolute top-1/2 -translate-y-1/2 -translate-x-1/2 {size_class} {shape_class} {team_color}{user_ring}{selected_class} cursor-pointer transition-transform")
                    style=format!("left: {left_pct}%")
                    title=tooltip
                    on:click=move |ev| {
                        ev.stop_propagation();
                        let current = selected_event.get_untracked();
                        if current == Some(idx) {
                            set_selected_event.set(None); // toggle off
                        } else {
                            set_selected_event.set(Some(idx)); // select
                        }
                    }
                />
            }
        }).collect_view()
    }}
