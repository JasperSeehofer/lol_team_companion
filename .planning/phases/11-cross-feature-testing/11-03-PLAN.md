**Server function** (defined BEFORE the component per CLAUDE.md rule 34):

```rust
#[server]
pub async fn get_analytics_data() -> Result<AnalyticsPayload, ServerFnError> {
    use crate::server::auth::AuthSession;
    use crate::server::db;
    use std::sync::Arc;
    use surrealdb::{engine::local::Db, Surreal};
    use crate::models::game_plan::AnalyticsPayload;

    let auth: AuthSession = leptos_axum::extract().await?;
    let user = auth.user.ok_or_else(|| ServerFnError::new("Not logged in"))?;

    let db = use_context::<Arc<Surreal<Db>>>()
        .ok_or_else(|| ServerFnError::new("No DB context"))?;

    // Return empty payload if user has no team (per CLAUDE.md rule 44 / D-10 no filters)
    let team_id = match db::get_user_team_id(&db, &user.id).await
        .map_err(|e| ServerFnError::new(e.to_string()))? {
        Some(id) => id,
        None => return Ok(AnalyticsPayload {
            tag_summaries: Vec::new(),
            plan_effectiveness: Vec::new(),
        }),
    };

    let (tag_summaries, plan_effectiveness) = db::get_analytics(&db, &team_id).await
        .map_err(|e| ServerFnError::new(e.to_string()))?;

    Ok(AnalyticsPayload { tag_summaries, plan_effectiveness })
}
```
