#[server]
pub async fn auto_detect_outcome(draft_id: String) -> Result<Option<String>, ServerFnError> {
    use crate::server::auth::AuthSession;
    use crate::server::db;
    use crate::server::riot;
    use std::sync::Arc;
    use surrealdb::{engine::local::Db, Surreal};

    let auth: AuthSession = leptos_axum::extract().await?;
    let user = auth.user.ok_or_else(|| ServerFnError::new("Not logged in"))?;

    let db = use_context::<Arc<Surreal<Db>>>()
        .ok_or_else(|| ServerFnError::new("No DB context"))?;

    // Get user's PUUID
    let puuid = match user.riot_puuid {
        Some(ref p) => p.clone(),
        None => return Ok(None), // No Riot account linked
    };

    // Get draft to find our champion picks
    let draft = match db::get_draft(&db, &draft_id).await.map_err(|e| ServerFnError::new(e.to_string()))? {
        Some(d) => d,
        None => return Ok(None),
    };

    // Extract our team's picked champions from draft actions
    let our_champs: Vec<String> = draft.actions.iter()
        .filter(|a| a.action_type == "pick" && a.team == "blue") // Adjust team logic as needed
        .filter_map(|a| a.champion.clone())
        .collect();

    if our_champs.is_empty() {

    // Fetch recent match IDs from Riot API (last 5 matches)

    for match_id in match_ids {
